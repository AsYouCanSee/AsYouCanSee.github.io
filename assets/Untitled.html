r2ave = []
choice = {'right', 'left', 'up', 'down'}
opp = {
    'right':'left',
    'left':'right',
    'up':'down',
    'down':'up'
}

np.random.seed(0)
for step in range(4, n+1):
    r2 = [0]
    for walk in range(int(w)):
        # since n = 50, we use 100x100 max array 
        a = np.full((2*n+1, 2*n+1), False)
        x = y = start
        dead = False
        moved = False
        prev = None
        for _ in range(step):
            a[x, y] = True
            
            if not moved:
                s = np.random.choice(list(choice))
                if s == 'right':
                    x += 1
                elif s == 'left':
                    x -= 1
                elif s == 'up':
                    y += 1
                elif s == 'down':
                    y -= 1
                
                moved = True
                prev = opp[s]
            else:
                avail = choice.difference({prev})
                s = np.random.choice(list(avail))
                if s == 'right' and not a[x+1, y]:
                    x += 1
                    prev = opp[s]
                elif s == 'left' and not a[x-1,y]:
                    x -= 1
                    prev = opp[s]
                elif s == 'up' and not a[x, y+1]:
                    y += 1
                    prev = opp[s]
                elif s == 'down' and not a[x, y-1]:
                    y -= 1
                    prev = opp[s]
                else:
                    dead = True
                    walk -= 1
                    break
                
        if not dead and len(r2) < w:
            r2.append((x - start)**2 + (y-start)**2)
    r2ave.append(np.mean(r2))

Since

\begin{gather*} \frac{\langle R_{n+1}^2 \rangle}{\langle R_n^2 \rangle} = \frac{(n+1)^{2\nu}}{n^{2\nu}} = \left( \frac{n+1}{n}\right)^{2\nu} = \left(1 + \frac{1}{n} \right)^{2\nu} \end{gather*}

Because $n$ is large then $1/n \ll 1$, and so by first order Taylor expansion, we have that

\begin{gather*} \boxed{\frac{\langle R_{n+1}^2 \rangle}{\langle R_n^2 \rangle} \approx 1 + 2\nu \frac{1}{n}} \end{gather*}

So since our our slope is $2\nu$, if follows that $\nu \approx 3/4$.
